017package org.fcrepo.indexer;
018
019import com.google.common.base.Supplier;
020import com.hp.hpl.jena.rdf.model.Model;
021import com.hp.hpl.jena.rdf.model.Property;
022import com.hp.hpl.jena.rdf.model.Resource;
023import org.apache.http.client.HttpClient;
024import org.apache.http.impl.client.DefaultHttpClient;
025import org.apache.http.impl.conn.PoolingClientConnectionManager;
026import org.fcrepo.kernel.utils.EventType;
027import org.slf4j.Logger;
028
029import javax.jms.JMSException;
030import javax.jms.Message;
031import javax.jms.MessageListener;
032import java.io.Reader;
033import java.util.Set;
034
035import static com.google.common.base.Suppliers.memoize;
036import static com.google.common.base.Throwables.propagate;
037import static com.hp.hpl.jena.rdf.model.ResourceFactory.createProperty;
038import static com.hp.hpl.jena.rdf.model.ResourceFactory.createResource;
039import static com.hp.hpl.jena.vocabulary.RDF.type;
040import static java.lang.Integer.MAX_VALUE;
041import static javax.jcr.observation.Event.NODE_REMOVED;
042import static org.fcrepo.kernel.RdfLexicon.REPOSITORY_NAMESPACE;
043import static org.slf4j.LoggerFactory.getLogger;
044
045/**
046 * MessageListener implementation that retrieves objects from the repository and
047 * invokes one or more indexers to index the content. documentation:
048 * https://wiki.duraspace.org/display/FF/Design+-+Messaging+for+Workflow
049 *
050 * @author Esm√© Cowles
051 * @author ajs6f
052 * @date Aug 19 2013
053 **/
054public class IndexerGroup implements MessageListener {
055
056    private static final Logger LOGGER = getLogger(IndexerGroup.class);
057
058    private String repositoryURL;
059
060    private Set<Indexer<Object>> indexers;
061
062    private HttpClient httpClient;
063
064    /**
065     * Identifier message header
066     */
067    static final String IDENTIFIER_HEADER_NAME = REPOSITORY_NAMESPACE
068            + "identifier";
069
070    /**
071     * Event type message header
072     */
073    static final String EVENT_TYPE_HEADER_NAME = REPOSITORY_NAMESPACE
074            + "eventType";
075
076    /**
077     * Type of event that qualifies as a removal.
078     */
079    static final String REMOVAL_EVENT_TYPE = REPOSITORY_NAMESPACE
080            + EventType.valueOf(NODE_REMOVED).toString();
081
082    public static final String INDEXER_NAMESPACE =
083        "http://fedora.info/definitions/v4/indexing#";
084
085    /**
086     * Indicates the transformation to use with this resource to derive indexing
087     * information.
088     */
089    public static final Property INDEXING_TRANSFORM_PREDICATE =
090        createProperty(INDEXER_NAMESPACE + "hasIndexingTransformation");
091
092    /**
093     * Indicates that a resource is indexable.
094     */
095    public static final Resource INDEXABLE_MIXIN =
096        createResource(INDEXER_NAMESPACE + "indexable");
097
098    private static final Reader EMPTY_CONTENT = null;
099
100    /**
101     * Default constructor.
102     **/
103    public IndexerGroup() {
104        LOGGER.debug("Creating IndexerGroup: {}", this);
105        final PoolingClientConnectionManager connMann =
106            new PoolingClientConnectionManager();
107        connMann.setMaxTotal(MAX_VALUE);
108        connMann.setDefaultMaxPerRoute(MAX_VALUE);
109        this.httpClient = new DefaultHttpClient(connMann);
110    }
111
112    /**
113     * Set repository URL.
114     **/
115    public void setRepositoryURL(final String repositoryURL) {
116        this.repositoryURL = repositoryURL;
117    }
118
119    /**
120     * Get repository URL.
121     **/
122    public String getRepositoryURL() {
123        return repositoryURL;
124    }
125
126    /**
127     * Set indexers for this group.
128     *
129     * @param indexers
130     */
131    public void setIndexers(final Set<Indexer<Object>> indexers) {
132        this.indexers = indexers;
133        LOGGER.debug("Using indexer complement: {} ", indexers);
134    }
135
136    /**
137     * Get indexers set for this group.
138     *
139     * @return indexers
140     */
141    public Set<Indexer<Object>> getIndexers() {
142        return indexers;
143    }
144
145    /**
146     * Set HttpClient for this group.  In the constructor a default is set
147     * but this allows it to be customized.
148     *
149     * @param client
150     */
151    public void setHttpClient(final HttpClient client) {
152        this.httpClient = client;
153    }
154
155    /**
156     * Gets the HttpClient used by this class.
157     *
158     * @return
159     */
160    public HttpClient getHttpClient() {
161        return this.httpClient;
162    }
163
164    /**
165     * Handle a JMS message representing an object update or deletion event.
166     **/
167    @Override
168    public void onMessage(final Message message) {
169        try {
170            LOGGER.debug("Received message: {}", message.getJMSMessageID());
171        } catch (final JMSException e) {
172            LOGGER.error("Received unintelligible message: {}", e);
173            propagate(e);
174        }
175        try {
176            final String pid;
177            // get pid and eventType from message
178            final String eventType =
179                message.getStringProperty(EVENT_TYPE_HEADER_NAME);
180            if (eventType.contains("PROPERTY")) {
181                // it seems the URL is for the property, not the node on which
182                // the property is set...
183                final String id = message.getStringProperty(IDENTIFIER_HEADER_NAME);
184                pid = id.substring(0, id.lastIndexOf('/'));
185            } else {
186                pid = message.getStringProperty(IDENTIFIER_HEADER_NAME);
187            }
188
189
190            LOGGER.debug("Discovered pid: {} in message.", pid);
191            LOGGER.debug("Discovered event type: {} in message.", eventType);
192
193            final Boolean removal = REMOVAL_EVENT_TYPE.equals(eventType);
194            LOGGER.debug("It is {} that this is a removal operation.", removal);
195            final String uri = getRepositoryURL() + pid;
196            final Supplier<Model> rdfr =
197                memoize(new RdfRetriever(uri, httpClient));
198            final Supplier<NamedFields> nfr =
199                memoize(new NamedFieldsRetriever(uri, httpClient, rdfr));
200            Boolean indexable = false;
201
202            if (!removal) {
203                final Model rdf = rdfr.get();
204                if (rdf.contains(createResource(uri), type, INDEXABLE_MIXIN)) {
205                    LOGGER.debug("Resource: {} retrieved with indexable type.",
206                            pid);
207                    indexable = true;
208                } else {
209                    LOGGER.debug(
210                            "Resource: {} retrieved without indexable type.",
211                            pid);
212                }
213            }
214
215            for (final Indexer<Object> indexer : getIndexers()) {
216                LOGGER.debug("Operating for indexer: {}", indexer);
217                Boolean hasContent = false;
218                Object content = EMPTY_CONTENT;
219                if (!removal && indexable) {
220                    switch (indexer.getIndexerType()) {
221                        case NAMEDFIELDS:
222                            LOGGER.debug(
223                                    "Retrieving named fields for: {}, (may be cached) to index to {}...",
224                                    pid, indexer);
225                            try  {
226                                content = nfr.get();
227                                hasContent = true;
228                            } catch (final AbsentTransformPropertyException e) {
229                                LOGGER.error("Failed to retrieve indexable content:"
230                                        + "could not find transform property!");
231                                hasContent = false;
232                            }
233                            break;
234                        case RDF:
235                            LOGGER.debug(
236                                    "Retrieving RDF for: {}, (may be cached) to index to {}...",
237                                    pid, indexer);
238                            content = rdfr.get();
239                            hasContent = true;
240                            break;
241                        default:
242                            hasContent = true;
243                            break;
244                    }
245                }
246
247                try {
248                    if (removal) {
249                        LOGGER.debug(
250                                "Executing removal of: {} to indexer: {}...",
251                                pid, indexer);
252                        indexer.remove(uri);
253                    } else {
254                        if (hasContent) {
255                            LOGGER.debug(
256                                    "Executing update of: {} to indexer: {}...",
257                                    pid, indexer);
258                            indexer.update(uri, content);
259                        } else if (indexable) {
260                            LOGGER.error(
261                                    "Received update for: {} but was unable to retrieve "
262                                            + "content for update to indexer: {}!",
263                                    pid, indexer);
264                        }
265                    }
266                } catch (final Exception e) {
267                    LOGGER.error("Error indexing {}: {}!", pid, e);
268                }
269            }
270
271        } catch (final JMSException e) {
272            LOGGER.error("Error processing JMS event!", e);
273        }
274    }
