050/**
051 * @author Michael Durbin
052 */
053public class IndexerGroupTest {
054
055    private IndexerGroup indexerGroup;
056
057    private String repoUrl;
058
059    @Mock
060    private HttpClient httpClient;
061
062    private Set<Indexer<Object>> indexers;
063
064    @Mock
065    private Indexer<Object> indexer;
066
067    @Before
068    public void setUp() {
069        initMocks(this);
070
071        repoUrl = "fake:fake";
072
073        indexers = new HashSet<Indexer<Object>>();
074        indexers.add(indexer);
075
076        indexerGroup = new IndexerGroup();
077        indexerGroup.setHttpClient(httpClient);
078        indexerGroup.setIndexers(indexers);
079        indexerGroup.setRepositoryURL(repoUrl);
080    }
081
082    @Test
083    public void testGetters() {
084        assertEquals(repoUrl, indexerGroup.getRepositoryURL());
085        assertEquals(indexers, indexerGroup.getIndexers());
086        assertEquals(httpClient, indexerGroup.getHttpClient());
087    }
088
089    @Test
090    public void testInvalidMessage() {
091        try {
092            indexerGroup.onMessage(createBrokenMessage());
093            fail("Broken message should result in an exception!");
094        } catch (Exception e) {
095        }
096    }
097
098    @Test
099    public void testNonIndexableObjectUpdateMessage() throws Exception {
100        String id = "/test";
101        indexerGroup.onMessage(createUnindexableMessage(REPOSITORY_NAMESPACE
102                + EventType.valueOf(NODE_ADDED).toString(), id));
103        verify(indexer, never()).update(anyString(), any());
104    }
105
106    @Test
107    public void testNamedFieldsIndexableObjectUpdateMessage() throws Exception {
108        when(indexer.getIndexerType()).thenReturn(Indexer.IndexerType.NAMEDFIELDS);
109        String id = "/test";
110        indexerGroup.onMessage(createIndexableMessage(REPOSITORY_NAMESPACE
111                + EventType.valueOf(NODE_ADDED).toString(), id));
112        verify(indexer, atLeastOnce()).update(anyString(), any());
113    }
114
115    @Test
116    public void testRDFIndexablePropertyUpdateMessage() throws Exception {
117        when(indexer.getIndexerType()).thenReturn(Indexer.IndexerType.RDF);
118        String id = "/test/dc:title";
119        indexerGroup.onMessage(createIndexablePropertyMessage(REPOSITORY_NAMESPACE
120                + EventType.valueOf(PROPERTY_CHANGED).toString(), id));
121        verify(indexer, atLeastOnce()).update(anyString(), any());
122    }
123
124    private Message createUnindexableMessage(String eventType, String identifier) throws Exception {
125        return createMockMessage(false, eventType, identifier, false, null, false);
126    }
127
128    private Message createIndexableMessage(String eventType, String identifier) throws Exception {
129        return createMockMessage(false, eventType, identifier, true, "default", false);
130    }
131
132    private Message createIndexablePropertyMessage(String eventType, String identifier) throws Exception {
133        return createMockMessage(false, eventType, identifier, true, "default", true);
134    }
135
136    private Message createBrokenMessage() throws Exception {
137        return createMockMessage(true, null, null, false, null, false);
138    }
139
140    /**
141     * Creates a mock message an updates the mock HTTPClient to respond to
142     * the expected request for that object.
143     */
144    private Message createMockMessage(boolean jmsExceptionOnGetMessage, String eventType, String identifier, boolean indexable, String indexerName, boolean property) throws Exception {
145        Message m = mock(Message.class);
146        if (jmsExceptionOnGetMessage) {
147            JMSException e = mock(JMSException.class);
148            when(m.getJMSMessageID()).thenThrow(e);
149        } else {
150            when(m.getJMSMessageID()).thenReturn("mocked-message-id");
151        }
152        if (eventType != null) {
153            when(m.getStringProperty(IndexerGroup.EVENT_TYPE_HEADER_NAME)).thenReturn(eventType);
154        }
155        if (identifier != null) {
156            when(m.getStringProperty(IndexerGroup.IDENTIFIER_HEADER_NAME)).thenReturn(identifier);
157            final HttpResponse r = mock(HttpResponse.class);
158            final StatusLine s = mock(StatusLine.class);
159            when(s.getStatusCode()).thenReturn(HttpStatus.SC_OK);
160            when(r.getStatusLine()).thenReturn(s);
161            final HttpEntity e = mock(HttpEntity.class);
162            when(e.getContent()).thenReturn(new ByteArrayInputStream(getIndexableTriples(property ? parentId(identifier) : identifier, indexable, indexerName).getBytes("UTF-8")));
163            when(r.getEntity()).thenReturn(e);
164            when(httpClient.execute(any(HttpUriRequest.class))).thenReturn(r);
165        }
166        return m;
167    }
168
169    private String parentId(String identifier) {
170        return identifier.substring(0, identifier.lastIndexOf('/'));
171    }
172
173    private String getIndexableTriples(String id, boolean indexable, String indexerName) {
174        return "\n" +
175                "<" + repoUrl + id + "> a <http://fedora.info/definitions/v4/rest-api#resource> , <http://fedora.info/definitions/v4/rest-api#object> ;\n" +
176                "\t<http://fedora.info/definitions/v4/repository#primaryType> \"nt:folder\"^^<http://www.w3.org/2001/XMLSchema#string> ;\n" +
177                (indexerName != null ? "\t<http://fedora.info/definitions/v4/indexing#hasIndexingTransformation> \"" + indexerName + "\"^^<http://www.w3.org/2001/XMLSchema#string> ;\n" : "") +
178                "\t<http://fedora.info/definitions/v4/repository#uuid> \"b1bfd6b8-b821-48c5-8eb9-05ef47e1b6e6\"^^<http://www.w3.org/2001/XMLSchema#string> ;\n" +
179                "\ta " + (indexable ? "<http://fedora.info/definitions/v4/indexing#indexable> , " : "") + "<http://fedora.info/definitions/v4/rest-api#resource> , <http://fedora.info/definitions/v4/rest-api#object> .\n";
180    }
