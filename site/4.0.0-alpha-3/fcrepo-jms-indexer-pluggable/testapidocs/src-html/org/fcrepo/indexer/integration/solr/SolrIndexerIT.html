018import static com.google.common.collect.ImmutableMap.of;
019import static java.lang.System.currentTimeMillis;
020import static java.lang.Thread.sleep;
021import static java.util.Arrays.asList;
022import static org.junit.Assert.assertEquals;
023import static org.slf4j.LoggerFactory.getLogger;
024
025import java.io.IOException;
026import java.util.Collection;
027import java.util.List;
028import org.apache.solr.client.solrj.SolrQuery;
029import org.apache.solr.client.solrj.SolrServer;
030import org.apache.solr.client.solrj.SolrServerException;
031import org.apache.solr.common.SolrDocument;
032import org.apache.solr.common.params.SolrParams;
033import org.fcrepo.indexer.NamedFields;
034import org.fcrepo.indexer.solr.SolrIndexer;
035import org.junit.Test;
036import org.junit.runner.RunWith;
037import org.slf4j.Logger;
038import org.springframework.beans.factory.annotation.Autowired;
039import org.springframework.test.context.ContextConfiguration;
040import org.springframework.test.context.junit4.SpringJUnit4ClassRunner;
041
042
043/**
044 * @author yecao
045 * @author ajs6f
046 * @date Nov 2013
047 */
048@RunWith(SpringJUnit4ClassRunner.class)
049@ContextConfiguration({"/spring-test/test-container.xml"})
050public class SolrIndexerIT {
051
052    @Autowired
053    private SolrIndexer solrIndexer;
054    @Autowired
055    private SolrServer server;
056
057    private static final Logger LOGGER = getLogger(SolrIndexerIT.class);
058
059    private static final long TIMEOUT = 15000;
060
061    private static final long TIME_TO_WAIT_STEP = 1000;
062
063    @Test
064    public void testUpdate() throws SolrServerException, IOException, InterruptedException {
065        doUpdate("123");
066    }
067
068    private void doUpdate(final String pid) throws SolrServerException, IOException, InterruptedException {
069        final Collection<String> values = asList(pid);
070        final NamedFields testContent = new NamedFields(of("id", values));
071        solrIndexer.update(pid, testContent);
072        final SolrParams query = new SolrQuery("id:" + pid);
073        List<SolrDocument> results = server.query(query).getResults();
074        Boolean success = results.size() == 1;
075        final Long start = currentTimeMillis();
076        while ((currentTimeMillis() - start < TIMEOUT) && !success) {
077            LOGGER.debug("Waiting for index record to appear...");
078            sleep(TIME_TO_WAIT_STEP);
079            LOGGER.debug("Checking for presence of appropriate index record...");
080            results = server.query(query).getResults();
081            success = results.size() == 1;
082        }
083        assertEquals("Didn't find the record that should have been created!",
084                pid, results.get(0).get("id"));
085    }
086
087    @Test
088    public void testRemove() throws IOException, SolrServerException, InterruptedException {
089        doUpdate("345");
090        solrIndexer.remove("345");
091        final SolrParams query = new SolrQuery("id:345");
092        List<SolrDocument> results = server.query(query).getResults();
093        Boolean success = results.size() == 0;
094        final Long start = currentTimeMillis();
095        while ((currentTimeMillis() - start < TIMEOUT) && !success) {
096            LOGGER.debug("Waiting for index record to appear...");
097            sleep(TIME_TO_WAIT_STEP);
098            LOGGER.debug("Checking for presence of appropriate index record...");
099            results = server.query(query).getResults();
100            success = results.size() == 0;
101        }
102        assertEquals("Found a record that should have been deleted!", 0,
103                results.size());
104    }
105
106}
