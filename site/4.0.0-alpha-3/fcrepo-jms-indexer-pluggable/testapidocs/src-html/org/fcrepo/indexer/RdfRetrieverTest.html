016package org.fcrepo.indexer;
017
018import static com.hp.hpl.jena.graph.NodeFactory.createURI;
019import static com.hp.hpl.jena.graph.Triple.create;
020import static com.hp.hpl.jena.rdf.model.ModelFactory.createDefaultModel;
021import static org.apache.http.HttpStatus.SC_NOT_FOUND;
022import static org.apache.http.HttpStatus.SC_OK;
023import static org.junit.Assert.assertTrue;
024import static org.mockito.Mockito.reset;
025import static org.mockito.Mockito.when;
026import static org.mockito.Mockito.any;
027import static org.mockito.MockitoAnnotations.initMocks;
028
029import java.io.ByteArrayInputStream;
030import java.io.IOException;
031import java.io.InputStream;
032import java.io.StringWriter;
033import org.apache.http.HttpEntity;
034import org.apache.http.HttpResponse;
035import org.apache.http.StatusLine;
036import org.apache.http.client.HttpClient;
037import org.apache.http.client.methods.HttpUriRequest;
038import org.junit.Before;
039import org.junit.Test;
040import org.mockito.Mock;
041import com.hp.hpl.jena.graph.Triple;
042import com.hp.hpl.jena.rdf.model.Model;
043
044
045public class RdfRetrieverTest {
046
047    private RdfRetriever testRetriever;
048
049    @Mock
050    private HttpClient mockClient;
051
052    @Mock
053    private HttpResponse mockResponse;
054
055    @Mock
056    private HttpEntity mockEntity;
057
058    @Mock
059    private StatusLine mockStatusLine;
060
061    private final Triple testTriple =
062            create(createURI("info:test"), createURI("info:test"),
063                    createURI("info:test"));
064
065    @Before
066    public void setUp() throws IOException {
067        initMocks(this);
068        when(mockClient.execute(any(HttpUriRequest.class))).thenReturn(
069                mockResponse);
070        when(mockResponse.getEntity()).thenReturn(mockEntity);
071        when(mockResponse.getStatusLine()).thenReturn(mockStatusLine);
072    }
073
074    @Test
075    public void testSimpleRetrieval() throws IOException {
076        final String testId = "testSimpleRetrieval";
077        final Model input = createDefaultModel();
078        input.add(input.asStatement(testTriple));
079        when(mockStatusLine.getStatusCode()).thenReturn(SC_OK);
080        try (StringWriter w = new StringWriter()) {
081            input.write(w, "N3");
082            try (
083                InputStream rdf =
084                    new ByteArrayInputStream(w.toString().getBytes())) {
085                when(mockEntity.getContent()).thenReturn(rdf);
086            }
087        }
088
089        testRetriever = new RdfRetriever(testId, mockClient);
090        final Model result = testRetriever.get();
091        assertTrue("Didn't find our test triple!", result.contains(result
092                .asStatement(testTriple)));
093
094    }
095
096    @Test(expected = RuntimeException.class)
097    public void testFailedRetrieval(){
098        final String testId = "testFailedRetrieval";
099        when(mockStatusLine.getStatusCode()).thenReturn(SC_NOT_FOUND);
100        new RdfRetriever(testId, mockClient).get();
101    }
102
103    @Test(expected = RuntimeException.class)
104    public void testOtherFailedRetrieval() throws IOException {
105        final String testId = "testFailedRetrieval";
106        when(mockStatusLine.getStatusCode()).thenReturn(SC_OK);
107        when(mockEntity.getContent()).thenThrow(new IOException());
108        new RdfRetriever(testId, mockClient).get();
109    }
110
111    @Test(expected = RuntimeException.class)
112    public void testYetOtherFailedRetrieval() throws IOException {
113        final String testId = "testFailedRetrieval";
114        reset(mockClient);
115        when(mockClient.execute(any(HttpUriRequest.class))).thenThrow(
116                new IOException());
117        when(mockStatusLine.getStatusCode()).thenReturn(SC_OK);
118        when(mockEntity.getContent()).thenThrow(new IOException());
119        new RdfRetriever(testId, mockClient).get();
120    }
121
122}
