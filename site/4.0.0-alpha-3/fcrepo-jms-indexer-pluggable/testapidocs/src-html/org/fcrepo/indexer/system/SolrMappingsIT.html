062@RunWith(SpringJUnit4ClassRunner.class)
063@ContextConfiguration({"/spring-test/test-container.xml"})
064public class SolrMappingsIT extends IndexingIT {
065
066    @Inject
067    private SolrServer indexServer;
068
069    private static final Logger LOGGER = getLogger(SolrMappingsIT.class);
070
071    private static final long TIMEOUT = 20000;
072
073    private static final long TIME_TO_WAIT_STEP = 1000;
074
075    @Inject
076    private SolrIndexer testSolrIndexer;
077
078    @Test
079    public void testOneResource() throws ClientProtocolException, IOException,
080        SolrServerException, InterruptedException {
081
082        LOGGER.debug("Creating indexable resource...");
083        final String uri = serverAddress + "testCollection/testOneResourcePid";
084        final HttpPost createRequest = new HttpPost(uri);
085        final String objectRdf =
086            "@prefix rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> ."
087                    + "@prefix dc:<http://purl.org/dc/elements/1.1/> ."
088                    + "@prefix indexing:<http://fedora.info/definitions/v4/indexing#>."
089                    + "<" + uri + ">  dc:title        \"500 Easy Microwave Meals\" ; "
090                    + "dc:creator      \"Yubulac Xorhorisa\" ; "
091                    + "dc:subject      <http://id.loc.gov/authorities/subjects/sh2012004374> ;"
092                    + "rdf:type  <http://fedora.info/definitions/v4/indexing#indexable> ;"
093                    + "indexing:hasIndexingTransformation \"default\".";
094
095        createRequest.setEntity(new StringEntity(objectRdf));
096        createRequest.addHeader("Content-Type", WebContent.contentTypeN3Alt1);
097        LOGGER.debug("Creating object with RDF:\n{}", objectRdf);
098        HttpResponse response = client.execute(createRequest);
099        assertEquals("Failed to create test resource!", SC_CREATED, response
100                .getStatusLine().getStatusCode());
101
102        LOGGER.debug("Waiting for our resource to be indexed...");
103        final String q = "title:\"500 Easy Microwave Meals\"";
104        LOGGER.debug("Checking for our record with query: {}", q);
105        final SolrQuery query = new SolrQuery(q);
106        List<SolrDocument> results = indexServer.query(query).getResults();
107        Boolean success = results.size() == 1;
108        Long start = currentTimeMillis();
109        while ((currentTimeMillis() - start < TIMEOUT) && !success) {
110            LOGGER.debug("Waiting for index record to appear...");
111            sleep(TIME_TO_WAIT_STEP);
112            LOGGER.debug("Checking for presence of appropriate index record...");
113            results = indexServer.query(query).getResults();
114            success = results.size() == 1;
115        }
116        assertTrue(
117                "Got other-than-one documents from index under our title!",
118                success);
119        LOGGER.debug("Discovered one record for {} in index.", uri);
120        assertEquals("Found wrong title in record!",
121                "500 Easy Microwave Meals", results.get(0).getFirstValue(
122                        "title"));
123
124        final HttpDelete deleteRequest = new HttpDelete(uri);
125        response = client.execute(deleteRequest);
126        assertEquals("Failed to delete test resource!", SC_NO_CONTENT, response
127                .getStatusLine().getStatusCode());
128        results = indexServer.query(query).getResults();
129        success = results.size() == 0;
130        start = currentTimeMillis();
131        while ((currentTimeMillis() - start < TIMEOUT) && !success) {
132            LOGGER.debug("Waiting for index record to disappear...");
133            sleep(TIME_TO_WAIT_STEP);
134            LOGGER.debug("Checking for absence of appropriate index record...");
135            results = indexServer.query(query).getResults();
136            success = results.size() == 0;
137        }
138        assertTrue(
139                "Found too many documents from index under our title!",
140                success);
141    }

143    /**
144     * Index the Dublin Core properties of a single resource, then remove the
145     * index record.
146     *
147     * @throws IOException
148     * @throws ClientProtocolException
149     * @throws SolrServerException
150     * @throws InterruptedException
151     */
152    //@Ignore("Waiting on https://www.pivotaltracker.com/story/show/61942638")
153    @Test
154    public void testOneResourceCustomMapping() throws ClientProtocolException, IOException,
155        SolrServerException, InterruptedException {
156
157        final String mappingUrl =
158            serverAddress + CONFIGURATION_FOLDER + "dc/indexingtest:book";
159
160        LOGGER.debug("Creating index mapping at URL: {}...", mappingUrl);
161
162        final HttpPost indexCreateRequest = new HttpPost(mappingUrl);
163        final HttpEntity indexMapping =
164            new StringEntity(Files.toString(new File(
165                    "target/test-classes/ldpath/dublin_core.ldpath"),
166                    defaultCharset()));
167        indexCreateRequest.setEntity(indexMapping);
168
169        HttpResponse response = client.execute(indexCreateRequest);
170        assertEquals("Failed to create index mapping!", SC_CREATED, response
171                .getStatusLine().getStatusCode());
172
173        LOGGER.debug("Creating indexable resource...");
174        final String uri = serverAddress + "testCollection/testOneResourceCustomMappingPid";
175        final HttpPost createRequest = new HttpPost(uri);
176        final String objectRdf =
177            "@prefix rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> ."
178                    + "@prefix dc:<http://purl.org/dc/elements/1.1/> ."
179                    + "@prefix indexing:<http://fedora.info/definitions/v4/indexing#>."
180                    + "<" + uri + ">  dc:title        \"500 Easy Microwave Meals\" ; "
181                    + "dc:creator      \"Yubulac Xorhorisa\" ; "
182                    + "dc:subject      <http://id.loc.gov/authorities/subjects/sh2012004374> ;"
183                    + "rdf:type  <http://fedora.info/definitions/v4/indexing#indexable> ;"
184                    + "rdf:type  <http://fedora.info/definitions/v4/indexingtest#book> ;"
185                    + "indexing:hasIndexingTransformation \"dc\".";
186
187        createRequest.setEntity(new StringEntity(objectRdf));
188        createRequest.addHeader("Content-Type", contentTypeN3Alt1);
189        LOGGER.debug("Creating object with RDF:\n{}", objectRdf);
190        response = client.execute(createRequest);
191        assertEquals("Failed to create test resource!", SC_CREATED, response
192                .getStatusLine().getStatusCode());
193
194        LOGGER.debug("Waiting for our resource to be indexed...");
195        final String q =
196            "subject:"
197                    + URLEncoder
198                            .encode("http://id.loc.gov/authorities/subjects/sh2012004374",
199                                    "UTF-8");
200        LOGGER.debug("Checking for our record with query: {}", q);
201        final SolrQuery query = new SolrQuery(q);
202        List<SolrDocument> results = indexServer.query(query).getResults();
203        Boolean success = results.size() == 1;
204        Long start = currentTimeMillis();
205        while ((currentTimeMillis() - start < TIMEOUT) && !success) {
206            LOGGER.debug("Waiting for index record to appear...");
207            sleep(TIME_TO_WAIT_STEP);
208            LOGGER.debug("Checking for presence of appropriate index record...");
209            results = indexServer.query(query).getResults();
210            LOGGER.debug("Found {} records with appropriate subject,", results
211                    .size());
212            success = results.size() == 1;
213        }
214        assertTrue(
215                "Got other-than-one documents from index under our title!",
216                success);
217        LOGGER.debug("Discovered one record for {} in index.", uri);
218        assertEquals("Found wrong title in record!",
219                "500 Easy Microwave Meals", results.get(0).getFirstValue(
220                        "title"));
221
222        final HttpDelete deleteRequest = new HttpDelete(uri);
223        response = client.execute(deleteRequest);
224        assertEquals("Failed to delete test resource!", SC_NO_CONTENT, response
225                .getStatusLine().getStatusCode());
226        results = indexServer.query(query).getResults();
227        success = results.size() == 0;
228        start = currentTimeMillis();
229        while ((currentTimeMillis() - start < TIMEOUT) && !success) {
230            LOGGER.debug("Waiting for index record to disappear...");
231            sleep(TIME_TO_WAIT_STEP);
232            LOGGER.debug("Checking for absence of appropriate index record...");
233            results = indexServer.query(query).getResults();
234            success = results.size() == 0;
235        }
236        assertTrue(
237                "Found too many documents from index under our title!",
238                success);
239    }
