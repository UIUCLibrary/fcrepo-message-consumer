169     /**
170      * Creates a mock message an updates the mock HTTPClient to respond to
171      * the expected request for that object.
172      */
173     private Message createMockMessage(final boolean jmsExceptionOnGetMessage,
174                                       final String eventType,
175                                       final String identifier,
176                                       final boolean indexable,
177                                       final String indexerName,
178                                       final boolean property) throws Exception {
179         final Message m = mock(Message.class);
180         if (jmsExceptionOnGetMessage) {
181             final JMSException e = mock(JMSException.class);
182             when(m.getJMSMessageID()).thenThrow(e);
183         } else {
184             when(m.getJMSMessageID()).thenReturn("mocked-message-id");
185         }
186         if (eventType != null) {
187             when(m.getStringProperty(IndexerGroup.EVENT_TYPE_HEADER_NAME)).thenReturn(eventType);
188         }
189         if (identifier != null) {
190             when(m.getStringProperty(IndexerGroup.IDENTIFIER_HEADER_NAME)).thenReturn(identifier);
191             mockContent(identifier, indexable, indexerName);
192         }
193         when(m.getStringProperty(IndexerGroup.BASE_URL_HEADER_NAME)).thenReturn("http://example.org:80");
194         return m;
195     }

197     private void mockContent(final String identifier,
198                              final boolean indexable,
199                              final String indexerName) throws Exception {
200         final CloseableHttpResponse r = mock(CloseableHttpResponse.class);
201         final StatusLine s = mock(StatusLine.class);
202         when(s.getStatusCode()).thenReturn(HttpStatus.SC_OK);
203         when(r.getStatusLine()).thenReturn(s);
204         final HttpEntity e = mock(HttpEntity.class);
205         when(e.getContent()).thenReturn(
206                 new ByteArrayInputStream(getIndexableTriples(identifier, indexable, indexerName).getBytes("UTF-8")));
207         when(r.getEntity()).thenReturn(e);
208         when(httpClient.execute(any(HttpUriRequest.class))).thenReturn(r);
209     }

211     private String getIndexableTriples(final String id, final boolean indexable, final String indexerName) {
212         return "\n" +
213                 "<" + repoUrl + id + "> a <http://fedora.info/definitions/v4/rest-api#resource> , " +
214                 "<http://fedora.info/definitions/v4/rest-api#object> ;\n" +
215                 "\t<http://fedora.info/definitions/v4/repository#primaryType> \"nt:folder\"^^<http://www.w3" +
216                 ".org/2001/XMLSchema#string> ;\n" +
217                 (indexerName != null ? "\t<http://fedora.info/definitions/v4/indexing#hasIndexingTransformation> \""
218                         + indexerName + "\"^^<http://www.w3.org/2001/XMLSchema#string> ;\n" : "") +
219                 "\t<http://fedora.info/definitions/v4/repository#uuid> " +
220                 "\"b1bfd6b8-b821-48c5-8eb9-05ef47e1b6e6\"^^<http://www.w3.org/2001/XMLSchema#string> ;\n" +
221                 "\ta " + (indexable ? "<http://fedora.info/definitions/v4/indexing#indexable> , " +
222                 "" : "") + "<http://fedora.info/definitions/v4/rest-api#resource> , " +
223                 "<http://fedora.info/definitions/v4/rest-api#object> .\n";
224     }
