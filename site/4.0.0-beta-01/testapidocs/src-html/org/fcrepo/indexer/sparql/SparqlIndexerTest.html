044public class SparqlIndexerTest {
045
046    private static final Logger LOGGER = getLogger(SparqlIndexerTest.class);
047
048    private SparqlIndexer testIndexer = new MockSparqlIndexer();
049
050    @Before
051    public void before() {
052        MockitoAnnotations.initMocks(this);
053    }
054
055    @Test
056    public void testGetIndexerType() {
057        assertEquals("Got wrong indexer type!", RDF, testIndexer.getIndexerType());
058        LOGGER.debug("Received correct indexer type.");
059    }
060
061    @Test
062    public void testRemoveSynch() {
063        testIndexer.removeSynch("info://obj-0");
064
065        final String cmd0 = "DELETE WHERE { <" + createURI("info://obj-0") + "> ?p ?o }";
066        final String cmd1 = "DELETE WHERE { <" + createURI("info://obj-0/fcr:content") + "> ?p ?o }";
067        final String cmd2 = "DELETE WHERE { <" + createURI("info://obj-0/child") + "> ?p ?o }";
068        Mockito.verify(updateRequest).add(cmd0);
069        Mockito.verify(updateRequest).add(cmd1);
070        Mockito.verify(updateRequest).add(cmd2);
071    }
072
073    @Test
074    public void testUpdateSynch() {
075        // TODO: This is a mere placeholder test to be further implemented later.
076        final Model model = ModelFactory.createDefaultModel();
077        testIndexer.updateSynch("", model);
078    }
079
080    @Mock
081    private QueryEngineHTTP queryEngineHTTP;
082
083    @Mock
084    private UpdateRequest updateRequest;
085
086    /**
087     * Test extension of SparqlIndexer to eliminate HTTP interactions.
088     */
089    private class MockSparqlIndexer extends SparqlIndexer {
090
091        protected QueryEngineHTTP buildQueryEngineHTTP(final String describeQuery) {
092            final Triple t0 = new Triple(createURI("info://sub"), createLiteral("p"), createURI("info://obj-0"));
093            final Triple t2 = new Triple(createURI("info://sub"),
094                                         createLiteral("p"),
095                                         createURI("info://obj-0/fcr:content"));
096            final Triple t1 = new Triple(createURI("info://sub"), createLiteral("p"), createURI("info://obj-1"));
097            final Triple t3 = new Triple(createURI("info://obj-0/child"),
098                                         createLiteral("p"),
099                                         createURI("info://obj-1"));
100
101            final Set<Triple> triples = new HashSet<>();
102            triples.add(t0);
103            triples.add(t1);
104            triples.add(t2);
105            triples.add(t3);
106
107            Mockito.when(queryEngineHTTP.execDescribeTriples()).thenReturn(triples.iterator());
108            return queryEngineHTTP;
109        }
110
111        protected UpdateRequest buildUpdateRequest() {
112            return updateRequest;
113        }
114    }
