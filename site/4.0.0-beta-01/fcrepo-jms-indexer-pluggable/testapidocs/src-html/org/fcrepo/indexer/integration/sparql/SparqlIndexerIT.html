016package org.fcrepo.indexer.integration.sparql;
017
018import java.io.StringReader;
019
020import javax.inject.Inject;
021
022import org.fcrepo.indexer.sparql.SparqlIndexer;
023import org.junit.runner.RunWith;
024import org.slf4j.Logger;
025import org.springframework.test.context.ContextConfiguration;
026import org.springframework.test.context.junit4.SpringJUnit4ClassRunner;
027
028import org.junit.Test;
029
030import static com.hp.hpl.jena.rdf.model.ModelFactory.createDefaultModel;
031import static java.lang.Thread.sleep;
032import static org.fcrepo.indexer.integration.sparql.SparqlIndexerITHelper.countQueryTriples;
033import static org.fcrepo.indexer.integration.sparql.SparqlIndexerITHelper.countDescribeTriples;
034import static org.junit.Assert.assertTrue;
035import static org.junit.Assert.assertEquals;
036import static org.slf4j.LoggerFactory.getLogger;
037
038/**
039 * @author Esm√© Cowles
040 *         Date: Aug 19, 2013
041 */
042@RunWith(SpringJUnit4ClassRunner.class)
043@ContextConfiguration({"/spring-test/test-container.xml"})
044public class SparqlIndexerIT {
045
046    private static final String uri = "http://example.com/sparqlIndexerTestURI";
047
048
049    private static final Logger LOGGER = getLogger(SparqlIndexerIT.class);
050
051    @Inject
052    private SparqlIndexer sparqlIndexer;
053
054    private static final String fooRDF =
055        "@prefix fcrepo: <http://fedora.info/definitions/v4/repository#> .\n" +
056        "@prefix fedora: <http://fedora.info/definitions/v4/repository/rest-api#> .\n" +
057        "<" + uri + ">\n" +
058        " fcrepo:hasChild <" + uri + "/barDS> ;\n" +
059        " fcrepo:hasParent <" + uri + "> ;\n" +
060        " fcrepo:uuid \"feb99ff2-455e-4e16-93a0-c0ae8d21b9ae\" .";
061
062    @Test
063    public void indexerTest() throws Exception {
064        // add triples
065        sparqlIndexer.update(uri, createDefaultModel().read(
066                new StringReader(fooRDF), "", "N3"));
067
068        waitForTriples(3);
069
070        // triples should be present in the triplestore
071        assertEquals("Triples should be present!", 3, countDescribeTriples(uri));
072
073        // SPARQL search should work
074        final String sparqlQuery =
075                "PREFIX  fcrepo: <http://fedora.info/definitions/v4/repository#> \n" +
076                        "SELECT  ?p \n" +
077                        "WHERE \n" +
078                        "{ ?p fcrepo:hasParent ?c }";
079        assertTrue("Triple should return from search!", countQueryTriples(sparqlQuery) > 0 );
080
081        // remove object
082        sparqlIndexer.remove(uri);
083
084        waitForTriples(0);
085
086        // triples should not be present in the triplestore
087        assertTrue("Triples should not be present!", countDescribeTriples(uri) == 0);
088    }
089
090    private static void waitForTriples(final int expectTriples) throws InterruptedException {
091        long elapsed = 0;
092        final long restingWait = 500;
093        final long maxWait = 15000; // 15 seconds
094
095        int count = countDescribeTriples(uri);
096        while ((count < expectTriples) && (elapsed < maxWait)) {
097            LOGGER.debug("Discovered {} triples, waiting for {}...", count, expectTriples);
098            sleep(restingWait);
099            count = countDescribeTriples(uri);
100
101            elapsed += restingWait;
102        }
103    }
104
105}
