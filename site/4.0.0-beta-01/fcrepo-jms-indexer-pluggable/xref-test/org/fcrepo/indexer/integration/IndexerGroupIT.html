61      @Test
62      public void testIndexerGroupUpdate() throws Exception {
63          doIndexerGroupUpdateTest(serverAddress + "updateTestPid");
64      }
65      private void doIndexerGroupUpdateTest(final String uri) throws Exception {
66          final HttpPut createRequest = new HttpPut(uri);
67          final String objectRdf =
68              "@prefix rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> ."
69                      + "@prefix indexing:<http://fedora.info/definitions/v4/indexing#>."
70                      + "<" + uri + ">  rdf:type  <http://fedora.info/definitions/v4/indexing#indexable> ;"
71                      + "indexing:hasIndexingTransformation \"default\".";
72  
73          createRequest.setEntity(new StringEntity(objectRdf));
74          createRequest.addHeader("Content-Type", contentTypeN3Alt1);
75  
76          final HttpResponse response = client.execute(createRequest);
77          assertEquals(201, response.getStatusLine().getStatusCode());
78          LOGGER.debug("Created object at: {}", uri);
79  
80          final Long start = currentTimeMillis();
81          synchronized (testIndexer) {
82              while (!testIndexer.receivedUpdate(uri)
83                      && (currentTimeMillis() - start < TIMEOUT)) {
84                  LOGGER.debug("Waiting for next notification from TestIndexer...");
85                  testIndexer.wait(1000);
86              }
87          }
88          assertTrue("Test indexer should have received an update message for " + uri + "!", testIndexer
89                  .receivedUpdate(uri));
90          LOGGER.debug("Received update at test indexer for identifier: {}", uri);
91  
92      }
93

122     @Test
123     public void testIndexerGroupReindex() throws Exception {
124         // create sample records
125         final String[] pids = { "a1", "a1/b1", "a1/b2", "a1/b1/c1" };
126         for ( String pid : pids ) {
127             doIndexerGroupUpdateTest(serverAddress + pid);
128         }
129 
130         // clear test indexer lists of updated records
131         testIndexer.clear();
132 
133         // reindex everything
134         indexerGroup.reindex();
135 
136         // records should be reindexed
137         synchronized (testIndexer) {
138             for ( String pid : pids ) {
139                 final String uri = serverAddress + pid;
140                 final Long start = currentTimeMillis();
141                 while (!testIndexer.receivedUpdate(uri) && (currentTimeMillis() - start < TIMEOUT)) {
142                     LOGGER.debug("Waiting for " + uri);
143                     testIndexer.wait(1000);
144                 }
145 
146                 assertTrue("Record should have been reindexed: " + uri,
147                         testIndexer.receivedUpdate(uri));
148             }
149         }
