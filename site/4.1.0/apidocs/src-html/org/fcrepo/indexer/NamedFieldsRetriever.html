016package org.fcrepo.indexer;
017
018import static com.google.common.base.Throwables.propagate;
019import static com.hp.hpl.jena.rdf.model.ResourceFactory.createResource;
020import static org.apache.http.HttpStatus.SC_OK;
021import static org.fcrepo.indexer.IndexerGroup.INDEXING_TRANSFORM_PREDICATE;
022import static org.slf4j.LoggerFactory.getLogger;
023
024import java.io.IOException;
025import java.io.InputStreamReader;
026import java.io.Reader;
027import java.lang.reflect.Type;
028import java.net.URI;
029import org.apache.http.HttpException;
030import org.apache.http.HttpResponse;
031import org.apache.http.client.HttpClient;
032import org.apache.http.client.methods.HttpGet;
033import org.slf4j.Logger;
034
035import com.google.common.base.Supplier;
036import com.google.gson.Gson;
037import com.google.gson.GsonBuilder;
038import com.google.gson.reflect.TypeToken;
039import com.hp.hpl.jena.rdf.model.Model;
040import com.hp.hpl.jena.rdf.model.RDFNode;
041
042/**
043 * Retrieves resources transformed into sets of named fields via LDPath.
044 * For use with indexers like Solr.
045 *
046 * @author ajs6f
047 * @since Dec 6, 2013
048 */
049public class NamedFieldsRetriever implements Supplier<NamedFields> {
050
051    private final URI uri;
052
053    private final HttpClient httpClient;
054
055    private final Supplier<Model> rdfr;
056
057    private Gson gson;
058
059    private static final Type typeToken = new TypeToken<NamedFields>() {}
060            .getType();
061
062    private static final Logger LOGGER = getLogger(NamedFieldsRetriever.class);
063
064    /**
065     * @param uri
066     * @param client
067     * @param rdfr Used to determine the transform to use with this indexing
068     *        step
069     */
070    public NamedFieldsRetriever(final URI uri, final HttpClient client,
071        final Supplier<Model> rdfr) {
072        this.uri = uri;
073        this.httpClient = client;
074        this.rdfr = rdfr;
075        final NamedFieldsDeserializer deserializer =
076            new NamedFieldsDeserializer();
077        this.gson =
078            new GsonBuilder().registerTypeAdapter(typeToken, deserializer)
079                    .create();
080        deserializer.setGson(gson);
081    }
082
083    @Override
084    public NamedFields get() {
085        LOGGER.debug("Retrieving RDF representation for: {}", uri);
086        try {
087            final Model rdf = rdfr.get();
088            if (!rdf.contains(createResource(uri.toString()), INDEXING_TRANSFORM_PREDICATE)) {
089                LOGGER.info(
090                        "Found no property locating LDPath transform for: {}, will not retrieve transformed content.",
091                        uri);
092                throw new AbsentTransformPropertyException(uri.toString());
093            }
094            final RDFNode indexingTransform =
095                rdf.listObjectsOfProperty(createResource(uri.toString()),
096                        INDEXING_TRANSFORM_PREDICATE).next();
097            final String transformKey =
098                indexingTransform.asLiteral().getString();
099
100            LOGGER.debug("Discovered transform key: {}", transformKey);
101            final HttpGet transformedResourceRequest =
102                new HttpGet(uri.toString() + "/fcr:transform/" + transformKey);
103            LOGGER.debug("Retrieving transformed resource from: {}",
104                    transformedResourceRequest.getURI());
105
106            final HttpResponse response =
107                httpClient.execute(transformedResourceRequest);
108            if (response.getStatusLine().getStatusCode() != SC_OK) {
109                throw new HttpException(response.getStatusLine().toString());
110            }
111            try (
112                Reader r =
113                    new InputStreamReader(response.getEntity().getContent(),
114                            "UTF8")) {
115                return gson.fromJson(r, typeToken);
116            }
117
118        } catch (IOException | HttpException e) {
119            throw propagate(e);
120        }
121    }
122
123}
